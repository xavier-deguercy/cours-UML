@startuml
skinparam style strictuml

actor Utilisateur
participant ": Bouton 1 " as B1
participant ": Bouton 2 " as B2
participant ": SimpleWatch" as SW
participant ": LCDDisplay" as LCD
participant ": Time" as T

== Accès au réglage des minutes ==
''loop
Utilisateur -> B1 : push()
B1 -> SW : notify()
SW -> LCD : blinkHours()
LCD --> Utilisateur : visualisation des Heures qui clignottes

Utilisateur -> B1 : push()
B1 -> SW : notify()
SW -> LCD : blinkHours(stop)
SW -> LCD : blinkMinutes()

LCD --> Utilisateur : visualisation des Minutes qui clignottes



== Incrémentation des minutes ==

loop Tant que l'utilisateur appuie sur le bouton 2
    Utilisateur -> B2 : push()
    B2 -> SW : notify()
    SW -> T : incrementMinutes()
    SW -> LCD : refresh()
    Utilisateur <-- LCD : NewTime

end

== Validation et Enregistrement ==

group Appui simultané (B1 + B2)
    group (2 buttons)
    Utilisateur -> B1 : push()
    Utilisateur -> B2 : push()
    B1 -> SW : notify()
    B2 -> SW : notify()
end
    SW -> T : commitNewTime()
    T --> SW : NewTime
        SW -> LCD : stopBlinking()
    Utilisateur <-- LCD : arretvisuelduclignottement

end

@enduml
